00047  { return true; }

00051 inline ebmpWORD FlipWORD( ebmpWORD in )

00054 inline ebmpDWORD FlipDWORD( ebmpDWORD in )
00055 {
00056  return ( ((in&0xFF000000)>>24) | ((in&0x000000FF)<<24) | 
00057           ((in&0x00FF0000)>>8 ) | ((in&0x0000FF00)<<8 )   );
