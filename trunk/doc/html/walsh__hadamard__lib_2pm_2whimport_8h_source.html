00135 /*
00136 Returns the distance between the given win and the segment from the given matrix
00137 that starts in (startRow, startCol). The segment's size will be as the win's size.
00138 */
00139 __inline distanceT findMatrixDistance(Matrix *mat, coordT startRow, coordT startCol, Matrix *win) {
00140         cellValueT *matPtr = matValPtr(mat) + startRow * matCols(mat) + startCol,
00141                    *winPtr = matValPtr(win);
00142         coordT rows = matRows(win), cols = matCols(win), col;
00143         coordT colsDif = matCols(mat) - cols;
00144         distanceT result = 0;
00145         while (rows--) {
00146                 col = cols;
00147                 while (col--) {
00148                         distanceT dif = *(matPtr++) - *(winPtr++);
00149                         result += dif * dif;
00150                 }
00151 
00152                 matPtr += colsDif;
00153         }
00154 
00155         return result;
00156 }
