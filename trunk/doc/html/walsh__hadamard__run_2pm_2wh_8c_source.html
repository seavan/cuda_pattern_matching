00066 The number of matrices created in the branch is (2 * depth) + 1:
00067  - The top matrix is reserved for the source matrix, and is not allocated.

00097

00111 #else

00113 #endif

00117                 || !pixelIDPtr)

00130                 depthT levelFromBottom = numOfMatrices - matInd;
00131

00133                 branch->horizontal[matInd - 1] = 1;

00350

00358 Computes, sets and returns the (y,x) pixel of matrix number matNo of the given basis
00359 in the given branch.
00360 */
00361 __inline cellValueT computePixel(Branch *branch, basisT basisInd, depthT matNo, coordT y, coordT x) {
00362 
00363         cellValueT secondVal, result;
00364         depthT nextMatNo = matNo - 1;
00365         basisT nodeID = branchNodeID(branch, nextMatNo, basisInd);
00366 
00367         // pixel is not updated for the current node - find value recursively.
00368         if (nodeID > branchPixelNodeID(branch, y, x)) {
00369                 // find second value according to horizontal/vertical
00370                 if (branchHorizontal(branch, nextMatNo))
00371                         secondVal = computePixel(branch, basisInd, nextMatNo, 
00372                                                                          y, (coordT)(x + branchDelta(branch, nextMatNo)));
00373                 else 
00374                         secondVal = computePixel(branch, basisInd, nextMatNo,
00375                                                                          (coordT)(y + branchDelta(branch, nextMatNo)), x);
00376 
00377                 // add/sub second value to/from base value according to sign
00378                 result = computePixel(branch, basisInd, nextMatNo, y, x)
00379                              + branchSign(branch, nextMatNo, basisInd) * secondVal;
00380 
00381                 branchPixelNodeID(branch, y, x) = nodeID;
00382         }
00383         // pixel is updated for the current node - just add/sub.
00384         else {
00385                 // find second value according to horizontal/vertical
00386                 if (branchHorizontal(branch, nextMatNo))
00387                         secondVal = branchMatVal(branch, nextMatNo, y, (coordT)(x + branchDelta(branch, nextMatNo)));
00388                 else 
00389                         secondVal = branchMatVal(branch, nextMatNo, (coordT)(y + branchDelta(branch, nextMatNo)), x);
00390 
00391                 // add/sub second value to/from base value according to sign
00392                 result = branchMatVal(branch, nextMatNo, y, x) + secondVal * branchSign(branch, nextMatNo, basisInd);
00393         }
00394 
00395         branchMatVal(branch, matNo, y, x) = result;

00420 /*
00421 Updates the current branch to contain the matrix branch of the given basis index.
00422 The leaf matrix of the updated branch is returned.
00423 */
00424 __inline Matrix * whMatrix(Branch *branch, basisT basisInd) {
00425         depthT climbLevels, level;
00426 
00427         // if no basis was created in the branch yet, the whole branch is created assuming
00428         // that the top matrix exists
00429         if (branchBasisInd(branch) == -1) {
00430                 level = 0;
00431                 climbLevels = branchDepth(branch);
00432         }
00433         // a basis was created in this branch before, compute the required starting level
00434         else {
00435                 // find num of levels to go up.
00436                 climbLevels = branchClimbLevels(branch, basisInd);
00437 
00438                 // find level in which the convolution should start
00439                 level = branchDepth(branch) - climbLevels; 
00440         }
00441 
00442         // replace required matrices in branch starting from the starting level
00443         for (; level < branchDepth(branch); level++, climbLevels--) {
00444                 
00445                 if (branchHorizontal(branch, level)) {
00446                         // add/sub the current matrix into the next one horizontally (using the current delta)
00447                         if (branchSign(branch, level, basisInd) == PLUS)
00448                                 addH(branchMat(branch, level), branchMat(branch, level + 1),
00449                                      branchDelta(branch, level));
00450                         else
00451                                 subH(branchMat(branch, level), branchMat(branch, level + 1),
00452                                      branchDelta(branch, level));
00453                 }
00454                 else {
00455                         // add/sub the current matrix into the next one vertically (using the current delta)
00456                         if (branchSign(branch, level, basisInd) == PLUS)
00457                                 addV(branchMat(branch, level), branchMat(branch, level + 1),
00458                                          branchDelta(branch, level));
00459                         else
00460                                 subV(branchMat(branch, level), branchMat(branch, level + 1),
00461                                          branchDelta(branch, level));
00462                 }
00463         }
00464 
00465         // set the new basis index.
00466         branchBasisInd(branch) = basisInd;
00467 
00468         // return the bottom matrix - the projection value of the given basis index
00469         return branchMat(branch, level);
00470 }

00576 /*****************************************************************************
00577  * Performs pattern matching on the given setup. The given rejectThresh      *
00578  * specifies the maximum mean difference (in pixels) between the pattern and *
00579  * a match.                                                                                                                                      *
00580  *                                                                                                                                                       *
00581  * After the execution of this function, the resulting matches array and the *
00582  * number of matches that were found are available in the setup, and can be  *
00583  * obtained using matches(setup) & numOfMatches(setup).                                          *
00584  *                                                                                                                                                       *
00585  * The pattern matching process is done using three methods, in this order:  *
00586  * Top Down - the whole image is projected on the WH kernels, starting with  *
00587  *            the lowest frequency kernel. Each such projection rejects some *
00588  *            of the suspected windows. After each projection, the percentage*
00589  *            of remaining suspected windows is measured.                                        *
00590  * Bottom Up - when the percentage of suspected windows goes under a certain *
00591  *             margin, the Top Down method is replaced by Bottom Up. In this *
00592  *             method, only the remaining suspected windows are projected,   *
00593  *             instead of the whole image. Here as well, each projection     *
00594  *             rejects some more suspected windows, effecting the percentage *
00595  *                         of remaining windows. The percentage margin for moving to     *
00596  *             Bottom Up is initially 10%, and it can be changed using the   *
00597  *             setMethodStartPercent() function.                                                     *
00598  * Direct Distance - when the percentage of suspected windows goes under     *
00599  *                   a second margin, the Bottom Up method is replaced by    *
00600  *                   Direct Distance. In this method, the eculidean distance *
00601  *                   between each of the remaining suspected windows and the *
00602  *                   pattern is computed, enabling the final rejection of    *
00603  *                   all the windows with distance above the threshold. The  *
00604  *                                   percentage margin for moving to Direct Distance is      *
00605  *                                   initially 2%, and it can be changed using the           *
00606  *                   setMethodStartPercent() function.                                           *
00607  *****************************************************************************/
00608 void whPatternMatch(WHSetup *setup, distanceT rejectThresh) {
00609 
00610         coordT y, x, projCols;
00611         unsigned int dif;
00612         matrixSizeT numOfWins, suspectedCount, total;
00613         Matrix *proj;
00614         Branch *branch = setup->branch;
00615         Match *currentMatch = matches(setup),
00616                   *baseMatch = matches(setup);
00617         cellValueT *projPtr;
00618         float suspectedPercent, percentFactor;
00619 
00620         booleanT performBottomUp = 0;
00621         unsigned int distanceShiftAmount = branch->depth;
00622         cellValueT *patternProj  = setup->patternProjections;
00623         basisT currentBasis = 1;
00624         distanceT threshold = rejectThresh * matSize(setup->patternImage); // don't square yet - for first scan
00625 
00626         // Perform initial projection of basis 0
00627         proj = whMatrix(branch, 0);
00628 
00629         numOfWins = matSize(proj);
00630         percentFactor = (float)100 / (float)numOfWins;
00631 
00632         // save the locations & distance of all suspected windows in the matches array.
00633         projPtr = matValPtr(proj);
00634         total = matSize(proj);
00635         projCols = matCols(proj);
00636         x = 0; y = 0;
00637         while (total) {
00638                 unsigned int count = min(total, BUFFER_SIZE);
00639                 cellValueT *bufferPtr = memcpy(buffer, projPtr, count << 2);
00640                 projPtr += count;
00641                 total -= count;
00642                 while (count--) {
00643                         if ((dif = abs(*bufferPtr - *patternProj)) <= threshold) {
00644                                 matchY(currentMatch) = y;
00645                                 matchX(currentMatch) = x;
00646                                 matchDistance(currentMatch++) = dif * dif;
00647                         }
00648                         bufferPtr++;
00649                         if (++x == projCols) {
00650                                 x = 0;
00651                                 y++;
00652                         }
00653                 }
00654         }
00655 
00656         suspectedCount = currentMatch - baseMatch;
00657         suspectedPercent = (float)suspectedCount * percentFactor;
00658         threshold *= rejectThresh * matSize(setup->patternImage); // square thresh - for next scans
00659         patternProj++;
00660 
00661         // Top Down phase: perform full projections for each basis
00662         // and udpate the "suspected" windows list according to the threshhold.
00663         while (suspectedPercent > setup->startBottomUpPercent && currentBasis < setup->numOfBasis) {
00664                 total = 0;
00665                 proj = whMatrix(branch, currentBasis);
00666                 currentMatch = baseMatch;
00667 
00668                 while (suspectedCount--) {
00669                         dif = matVal(proj, matchY(currentMatch), matchX(currentMatch)) - *patternProj;
00670                         if ((matchDistance(currentMatch) += dif * dif) <= threshold)
00671                                 baseMatch[total++] = *currentMatch;
00672                         currentMatch++;
00673                 }
00674 
00675                 suspectedCount = total;
00676                 suspectedPercent = (float)suspectedCount * percentFactor;
00677                 currentBasis++;
00678                 patternProj++;
00679         }
00680 
00681         // check whether Bottom Up phase is needed, or maybe we should go directly to the 
00682         // Direct Distance phase.
00683         performBottomUp = suspectedPercent > setup->startDistancePercent &&
00684                                           currentBasis < setup->numOfBasis && suspectedCount;
00685 
00686         // when Top Down phase has ended, the nodeID of the last updated leaf should be updated
00687         // for all pixels, as a preparation for the Bottom Up phase (if necessary)
00688         if (performBottomUp)
00689                 updateNodeID(branch);
00690 
00691         // Bottom Up phase: Scan basis, for each one find the projections of the "suspected" windows
00692         // and udpate the "suspected" windows list according to the threshhold.
00693         while (performBottomUp) {
00694                 total = 0;
00695                 currentMatch = baseMatch;
00696 
00697                 while (suspectedCount--) {
00698                         // increase lower bound by the distance of the current basis
00699                         dif = whPixel(branch, currentBasis, matchY(currentMatch), matchX(currentMatch)) - *patternProj;
00700                         // if lower bound is under threshold, save the current win in the updated win list
00701                         if ((matchDistance(currentMatch) += dif * dif) <= threshold)
00702                                 baseMatch[total++] = *currentMatch;
00703 
00704                         currentMatch++;
00705                 }
00706 
00707                 suspectedCount = total;
00708                 suspectedPercent = (float)suspectedCount * percentFactor;
00709                 currentBasis++;
00710                 patternProj++;
00711                 performBottomUp = suspectedPercent > setup->startDistancePercent &&
00712                                                   currentBasis < setup->numOfBasis && suspectedCount;
00713         }
00714 
00715         // Direct Distance phase: compute the direct distance of each suspected window and update
00716         // the suspected windows list accordingly.
00717         if (suspectedCount) {
00718                 total = 0;
00719                 threshold /= matSize(setup->patternImage);
00720                 currentMatch  = baseMatch;
00721 
00722                 while (suspectedCount--) {
00723                         // if lower bound is under threshold, save the current win in the updated win list
00724                         if ((matchDistance(currentMatch) = findMatrixDistance(setup->sourceImage,
00725                                                                                                                               matchY(currentMatch),
00726                                                                                                                                   matchX(currentMatch),
00727                                                                                                                                   setup->patternImage))
00728                                         <= threshold)
00729                                 baseMatch[total++] = *currentMatch;
00730 
00731                         currentMatch++;
00732                 }
00733 
00734                 suspectedCount = total;
00735         }
00736 
00737         numOfMatches(setup) = suspectedCount;
00738 }
