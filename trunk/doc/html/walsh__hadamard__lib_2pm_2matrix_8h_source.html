00057 /*
00058 Returns the distance between the given win and the segment from the given matrix
00059 that starts in (startRow, startCol). The segment's size will be as the win's size.
00060 */
00061 __inline distanceT findMatrixDistance(Matrix *mat, coordT startRow, coordT startCol, Matrix *win) {
00062         cellValueT *matPtr = matValPtr(mat) + startRow * matCols(mat) + startCol,
00063                    *winPtr = matValPtr(win);
00064         coordT rows = matRows(win), cols = matCols(win), col;
00065         coordT colsDif = matCols(mat) - cols;
00066         distanceT result = 0;
00067         while (rows--) {
00068                 col = cols;
00069                 while (col--) {
00070                         distanceT dif = *(matPtr++) - *(winPtr++);
00071                         result += dif * dif;
00072                 }
00073 
00074                 matPtr += colsDif;
00075         }
00076 
00077         return result;
00078 }
