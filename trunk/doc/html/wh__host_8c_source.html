00065 /*
00066 Allocates and returns an Branch of the given depth with the given matrix sizes.
00067 The number of matrices created in the branch is (2 * depth) + 1:
00068  - The top matrix is reserved for the source matrix, and is not allocated.
00069  - each unit of depth allocates two matrices, one for the horizontal convolution and one
00070    for the vertical convolution.
00071 The basis index of the branch is initialized to -1.
00072 */
00073 DEVICE Branch *allocBranch(coordT rows, coordT cols, depthT depth, basisT numOfBasis)
00074 {
00075         Branch *branch;
00076         coordT delta, row;

00078         basisT numOfLeaves = (basisT)pow(2, numOfMatrices - 1);
00079         basisT divider = 2;
00080         depthT matInd;
00081         signT *signPtr;
00082         basisT *nodeIDPtr, *pixelIDPtr;
00083         depthT *climbPtr = (depthT *)1; // init with dummy value
00084         basisT i = 0, j = 0; // init with dummy values
00085 
00086         // allocate branch
00087         branch = (Branch *)malloc(sizeof(Branch));
00088         if (!branch)
00089                 exitWithError("ERROR in allocBranch: can't allocate branch.");
00090 
00091         // allocate matrices
00092         branch->matrices = (Matrix **)malloc(numOfMatrices * sizeof(Matrix *));
00093         if (!branch->matrices)
00094                 exitWithError("ERROR in allocBranch: can't allocate matrices array mapping.");
00095 
00096         branch->depth = depth << 1;
00097         branch->basisInd = -1;
00098 
00099         // allocate branch data
00100         branch->delta = (coordT *)malloc(branch->depth * sizeof(coordT));
00101         branch->horizontal = (booleanT *)malloc(branch->depth * sizeof(booleanT));
00102         branch->sign = (signT **)malloc(branch->depth * sizeof(signT *));
00103         signPtr = (signT *)malloc(branch->depth * numOfBasis * sizeof(signT));
00104         branch->nodeID = (basisT **)malloc(numOfMatrices * sizeof(basisT *));
00105         nodeIDPtr = (basisT *)malloc(numOfMatrices * numOfBasis * sizeof(basisT));
00106         branch->pixelNodeIDRowsPtr = (basisT **)malloc(rows * sizeof(basisT *));
00107         branch->pixelNodeIDValues = pixelIDPtr = (basisT *)malloc(rows * cols * sizeof(basisT));
00108 
00109 #ifdef MAX_BASIS_1024
00110         branch->climbLevels = (depthT **)malloc(numOfBasis * sizeof(depthT *));
00111         climbPtr = (depthT *)malloc(numOfBasis * numOfBasis * sizeof(depthT));
00112 #else
00113         branch->climbLevels = (depthT *)malloc(numOfBasis * sizeof(depthT *));
00114 #endif
00115         
00116         if (!branch->delta || !branch->horizontal || !branch->sign || !branch->climbLevels
00117                 || !branch->nodeID || !branch->pixelNodeIDRowsPtr || !signPtr || !climbPtr || !nodeIDPtr
00118                 || !pixelIDPtr)
00119                 exitWithError("ERROR in allocBranch: can't allocate branch data.");
00120 
00121         // create pixel node ID 2-d array (uninitialized)
00122         for (row = 0; row < rows ; row++, pixelIDPtr += cols)
00123                 branch->pixelNodeIDRowsPtr[row] = pixelIDPtr;
00124 
00125         branch->matrices[0] = NULL;
00126         
00127         // allocate matrices, where each matrix has 2^i less rows than the previous matrix
00128         // matrices[0] is not allocated at this point.
00129         // Also sets the appropriate delta, horizontal & sign array for each level.
00130         for (matInd = 1, delta = 1; matInd < numOfMatrices; matInd += 2, delta <<= 1) {
00131                 depthT levelFromBottom = numOfMatrices - matInd;
00132 
00133                 // fill horizontal & delta array
00134                 branch->horizontal[matInd - 1] = 1;
00135                 branch->horizontal[matInd] = 0;
00136                 branch->delta[matInd - 1] = delta;
00137                 branch->delta[matInd] = delta;
00138 
00139                 // fill sign 2-d array
00140                 branch->sign[matInd - 1] = signPtr;
00141                 branch->sign[matInd] = signPtr + numOfBasis;
00142                 i = 0;
00143                 while (i < numOfBasis) {
00144                         *signPtr = getSign(i, levelFromBottom);
00145                         *(signPtr + numOfBasis) = getSign(i, (depthT)(levelFromBottom - 1));
00146                         signPtr++;
00147                         i++;
00148                 }
00149                 signPtr += numOfBasis;
00150 
00151                 // fill matrices array
00152                 branch->matrices[matInd] = allocMatrix(rows, cols - delta);
00153                 branch->matrices[matInd + 1] = allocMatrix(rows - delta, cols - delta);
00154                 rows -= delta;
00155                 cols -= delta;
00156         }
00157         
00158         // fill node ID 2-d array
00159 
00160         branch->nodeID[0] = nodeIDPtr;
00161         
00162         for (i = 0; i < numOfBasis; i++) // init node ID of first level (level 0)
00163                 *(nodeIDPtr++) = 0;
00164 
00165         for (matInd = 1, divider = 2; matInd < numOfMatrices; matInd++, divider *= 2, nodeIDPtr += numOfBasis) {
00166                 basisT limit = numOfLeaves / divider;
00167                 basisT addedValue = 1;
00168                 booleanT exit = 0;
00169                 basisT basisInd = 0;
00170 
00171                 branch->nodeID[matInd] = nodeIDPtr;
00172 
00173                 for (i = 0; i < divider && !exit; i++) {
00174                         for (j = 0; j < limit && !exit; j++, basisInd++) {
00175                                 exit = (basisInd >= numOfBasis);
00176                                 if (!exit)
00177                                         branch->nodeID[matInd][basisInd] = branch->nodeID[matInd - 1][basisInd] + addedValue;
00178                         }
00179                         addedValue = limit * 2 - addedValue + 1;
00180                 }
00181         }
00182 
00183         // fill climb levels array
00184         i = 0;
00185         while (i < numOfBasis) {
00186 
00187 #ifdef MAX_BASIS_1024
00188                 branch->climbLevels[i] = climbPtr;  // fill 2-d array
00189                 j = 0;
00190                 while (j < numOfBasis) {
00191                         *(climbPtr++) = getClimbLevels(i, j);
00192                         j++;
00193                 }
00194 #else
00195                 branch->climbLevels[i] = getClimbLevels(i, i + 1); // fill 1-d array
00196 #endif
00197 
00198                 i++;
00199         }
00200 
00201         return(branch);
00202 }

00238 __inline void addH(Matrix *sourceMat, Matrix *destMat, coordT delta) {
00239         cellValueT *sourcePtr = matValPtr(sourceMat),
00240                        *destPtr = matValPtr(destMat);

00242                    sourceCols = matCols(sourceMat);

00245 
00246         while (total) {
00247                 unsigned int count = min(total, maxBuffer);
00248                 cellValueT *bufferPtr = (cellValueT*)memcpy(buffer, sourcePtr, count << 2);
00249                 sourcePtr += count;
00250                 total -= count;
00251                 while (count) {
00252                         coordT col = destCols;
00253                         while (col--)
00254                                 *(destPtr++) = *(bufferPtr) + *(bufferPtr++ + delta);
00255                         bufferPtr += delta;
00256                         count -= sourceCols;
00257                 }
00258         }

00343 /*
00344 Finds the number of levels to climb in the wh tree from the source basis index to the dest
00345 basis index in order to find the lowest common matrix of the two given basis.
00346 */
00347 __inline depthT getClimbLevels(basisT sourceBasisInd, basisT destBasisInd) {
00348         // XOR the current and new basis indice
00349         int xorInd = sourceBasisInd ^ destBasisInd;
00350         depthT result;
00351 
00352         // find num of levels to go up.
00353         for (result = 0; xorInd > 0; xorInd >>= 1, result++);

00355         return result;

00358 /*
00359 Computes, sets and returns the (y,x) pixel of matrix number matNo of the given basis
00360 in the given branch.
00361 */
00362 __inline cellValueT computePixel(Branch *branch, basisT basisInd, depthT matNo, coordT y, coordT x) {
00363 
00364         cellValueT secondVal, result;
00365         depthT nextMatNo = matNo - 1;
00366         basisT nodeID = branchNodeID(branch, nextMatNo, basisInd);
00367 
00368         // pixel is not updated for the current node - find value recursively.
00369         if (nodeID > branchPixelNodeID(branch, y, x)) {
00370                 // find second value according to horizontal/vertical
00371                 if (branchHorizontal(branch, nextMatNo))
00372                         secondVal = computePixel(branch, basisInd, nextMatNo, 
00373                                                                          y, (coordT)(x + branchDelta(branch, nextMatNo)));
00374                 else 
00375                         secondVal = computePixel(branch, basisInd, nextMatNo,
00376                                                                          (coordT)(y + branchDelta(branch, nextMatNo)), x);
00377 
00378                 // add/sub second value to/from base value according to sign
00379                 result = computePixel(branch, basisInd, nextMatNo, y, x)
00380                              + branchSign(branch, nextMatNo, basisInd) * secondVal;
00381 
00382                 branchPixelNodeID(branch, y, x) = nodeID;
00383         }
00384         // pixel is updated for the current node - just add/sub.
00385         else {
00386                 // find second value according to horizontal/vertical
00387                 if (branchHorizontal(branch, nextMatNo))
00388                         secondVal = branchMatVal(branch, nextMatNo, y, (coordT)(x + branchDelta(branch, nextMatNo)));
00389                 else 
00390                         secondVal = branchMatVal(branch, nextMatNo, (coordT)(y + branchDelta(branch, nextMatNo)), x);
00391 
00392                 // add/sub second value to/from base value according to sign
00393                 result = branchMatVal(branch, nextMatNo, y, x) + secondVal * branchSign(branch, nextMatNo, basisInd);
00394         }
00395 
00396         branchMatVal(branch, matNo, y, x) = result;
00397         return result;
00398 }

00421 /*
00422 Updates the current branch to contain the matrix branch of the given basis index.
00423 The leaf matrix of the updated branch is returned.
00424 */
00425 __inline Matrix * whMatrix(Branch *branch, basisT basisInd) {
00426         depthT climbLevels, level;
00427 
00428         // if no basis was created in the branch yet, the whole branch is created assuming
00429         // that the top matrix exists
00430         if (branchBasisInd(branch) == -1) {
00431                 level = 0;
00432                 climbLevels = branchDepth(branch);
00433         }
00434         // a basis was created in this branch before, compute the required starting level
00435         else {
00436                 // find num of levels to go up.
00437                 climbLevels = branchClimbLevels(branch, basisInd);
00438 
00439                 // find level in which the convolution should start
00440                 level = branchDepth(branch) - climbLevels; 
00441         }
00442 
00443         // replace required matrices in branch starting from the starting level
00444         for (; level < branchDepth(branch); level++, climbLevels--) {
00445                 
00446                 if (branchHorizontal(branch, level)) {
00447                         // add/sub the current matrix into the next one horizontally (using the current delta)
00448                         if (branchSign(branch, level, basisInd) == PLUS)
00449                                 addH(branchMat(branch, level), branchMat(branch, level + 1),
00450                                      branchDelta(branch, level));
00451                         else
00452                                 subH(branchMat(branch, level), branchMat(branch, level + 1),
00453                                      branchDelta(branch, level));
00454                 }
00455                 else {
00456                         // add/sub the current matrix into the next one vertically (using the current delta)
00457                         if (branchSign(branch, level, basisInd) == PLUS)
00458                                 addV(branchMat(branch, level), branchMat(branch, level + 1),
00459                                          branchDelta(branch, level));
00460                         else
00461                                 subV(branchMat(branch, level), branchMat(branch, level + 1),
00462                                          branchDelta(branch, level));
00463                 }
00464         }
00465 
00466         // set the new basis index.
00467         branchBasisInd(branch) = basisInd;
00468 
00469         // return the bottom matrix - the projection value of the given basis index
00470         return branchMat(branch, level);
00471 }
